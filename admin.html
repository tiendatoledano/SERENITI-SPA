ACTUALIZAR CHECKBOXES DE DÍAS const diasContainer = document.getElementById('diasContainer'); if (diasContainer) { const diasGuardados = horariosConfig.dias || []; console.log('📋 Días guardados:', diasGuardados); // Obtener todos los checkboxes dentro del container const checkboxes = diasContainer.querySelectorAll('input[type="checkbox"]'); console.log(`📋 Encontrados ${checkboxes.length} checkboxes de días`); checkboxes.forEach(cb => { // Marcar/desmarcar según los días guardados const deberiaEstarChecked = diasGuardados.includes(cb.value); cb.checked = deberiaEstarChecked; // Actualizar estilo visual del label padre const label = cb.closest('label'); if (label) { if (deberiaEstarChecked) { label.classList.add('checked'); } else { label.classList.remove('checked'); } } // Log para depuración console.log(` Día ${cb.value}: ${cb.checked ? '✅ seleccionado' : '❌ no seleccionado'}`); }); // Agregar eventos para cambio visual (solo una vez) if (!diasContainer._listenersAdded) { checkboxes.forEach(cb => { cb.addEventListener('change', function() { const label = this.closest('label'); if (label) { if (this.checked) { label.classList.add('checked'); } else { label.classList.remove('checked'); } } console.log(`📋 Día ${this.value} ${this.checked ? '✅ seleccionado' : '❌ deseleccionado'}`); }); }); diasContainer._listenersAdded = true; console.log('✅ Eventos de cambio agregados a los checkboxes'); } } else { console.warn('⚠️ No se encontró el contenedor #diasContainer'); } // 2. ACTUALIZAR LISTA DE HORARIOS const horariosList = document.getElementById('horariosList'); if (horariosList) { const horarios = horariosConfig.horarios || []; console.log(`📋 Horarios guardados: ${horarios.length} horarios`); if (horarios.length === 0) { horariosList.innerHTML = 'No hay horarios configurados'; } else { horariosList.innerHTML = horarios.map(h => `${h} ` ).join(''); } } else { console.warn('⚠️ No se encontró el contenedor #horariosList'); } console.log('✅ UI de horarios renderizada'); } window.eliminarHorario = function(horario) { console.log(`🗑️ Eliminando horario: ${horario}`); horariosConfig.horarios = horariosConfig.horarios.filter(h => h !== horario); renderHorariosUI(); showToast(`Horario ${horario} eliminado`, 'warning'); }; function agregarHorario() { const input = document.getElementById('nuevoHorario'); const valor = input.value; if (!valor) { showToast('Seleccione un horario', 'error'); return; } if (!horariosConfig.horarios.includes(valor)) { horariosConfig.horarios.push(valor); horariosConfig.horarios.sort(); renderHorariosUI(); input.value = ''; showToast(`Horario ${valor} agregado`); } else { showToast('El horario ya existe', 'error'); } } async function guardarHorarios() { try { console.log('🔄 Iniciando guardado de horarios...'); // Obtener días seleccionados de los checkboxes const dias = []; document.querySelectorAll('#diasContainer input[type="checkbox"]:checked').forEach(cb => { dias.push(cb.value); }); console.log('📋 Días seleccionados:', dias); if (dias.length === 0) { showToast('⚠️ Selecciona al menos un día', 'warning'); return; } // Obtener horarios actuales de la lista const horarios = horariosConfig.horarios || []; console.log('📋 Horarios actuales:', horarios); // Obtener inicio y fin de los inputs const inicioInput = document.getElementById('horaInicio'); const finInput = document.getElementById('horaFin'); let inicio = 8; let fin = 20; if (inicioInput && inicioInput.value) { inicio = parseInt(inicioInput.value.split(':')[0]); if (isNaN(inicio)) inicio = 8; } if (finInput && finInput.value) { fin = parseInt(finInput.value.split(':')[0]); if (isNaN(fin)) fin = 20; } console.log(`📋 Rango horario: ${inicio}:00 - ${fin}:00`); // Preparar datos para enviar const data = { dias: dias, horarios: horarios, inicio: inicio, fin: fin }; console.log('📤 Enviando datos al servidor:', JSON.stringify(data, null, 2)); // Enviar al servidor const response = await apiFetch(HORARIOS_API, { method: 'PUT', body: JSON.stringify(data) }); if (response.ok) { const result = await response.json(); console.log('✅ Respuesta del servidor:', result); // Actualizar la configuración local con los datos guardados horariosConfig = data; renderHorariosUI(); showToast('✅ Horarios guardados correctamente'); // Si existe la función renderCalendar en el index, recargar if (typeof renderCalendar === 'function') { console.log('🔄 Recargando calendario...'); renderCalendar(); } } else { const error = await response.json(); console.error('❌ Error del servidor:', error); showToast('❌ Error: ' + (error.error || 'Desconocido'), 'error'); } } catch(e) { console.error('❌ Error guardando horarios:', e); showToast('❌ Error al guardar horarios: ' + e.message, 'error'); } } async function guardarRangoHorario() { try { console.log('🔄 Guardando rango horario...'); const inicioStr = document.getElementById('horaInicio').value; const finStr = document.getElementById('horaFin').value; if (!inicioStr || !finStr) { showToast('⚠️ Selecciona ambas horas', 'error'); return; } const inicio = parseInt(inicioStr.split(':')[0]); const fin = parseInt(finStr.split(':')[0]); if (isNaN(inicio) || isNaN(fin)) { showToast('⚠️ Horas inválidas', 'error'); return; } if (inicio >= fin) { showToast('⚠️ La hora de inicio debe ser menor que la de cierre', 'error'); return; } // Generar horarios automáticamente const horarios = []; for (let h = inicio; h < fin; h++) { horarios.push(`${String(h).padStart(2, '0')}:00`); } console.log(`📋 Horarios generados: ${horarios.length} horarios`); // Obtener días seleccionados const dias = []; document.querySelectorAll('#diasContainer input[type="checkbox"]:checked').forEach(cb => { dias.push(cb.value); }); console.log('📋 Días seleccionados:', dias); if (dias.length === 0) { showToast('⚠️ Selecciona al menos un día', 'warning'); return; } const data = { dias: dias, horarios: horarios, inicio: inicio, fin: fin }; console.log('📤 Enviando datos al servidor:', JSON.stringify(data, null, 2)); const response = await apiFetch(HORARIOS_API, { method: 'PUT', body: JSON.stringify(data) }); if (response.ok) { const result = await response.json(); console.log('✅ Respuesta del servidor:', result); horariosConfig = data; renderHorariosUI(); showToast('✅ Rango horario guardado'); if (typeof renderCalendar === 'function') { console.log('🔄 Recargando calendario...'); renderCalendar(); } } else { const error = await response.json(); console.error('❌ Error del servidor:', error); showToast('❌ Error: ' + (error.error || 'Desconocido'), 'error'); } } catch(e) { console.error('❌ Error guardando rango horario:', e); showToast('❌ Error al guardar rango horario: ' + e.message, 'error'); } }
